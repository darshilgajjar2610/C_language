int main() {
    int n = 5; // Number of rows

    for (int i = 1; i <= n; i++) 
    {
        // Print spaces
        for (int j = 1; j <= n - i; j++) 
        {
            printf("   ");
        }

        // Print numbers in descending order
        for (int k = n; k >= n - i + 1; k--) 
        {
            printf("%d  ", k);
            printf("%d  ", i);


        }

        printf("\n");
    }

    return 0;
}
